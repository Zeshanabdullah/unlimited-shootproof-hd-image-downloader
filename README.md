### Python Example
This example demonstrates how to use the `requests` library to download an image from ShootProof HD Image Downloader.

python
import requests
import os

def download_image(image_url, save_dir='downloads'):
    try:
        response = requests.get(image_url, stream=True)
        response.raise_for_status()  # Raise an error for bad responses
        
        if not os.path.exists(save_dir):
            os.makedirs(save_dir)
        
        file_name = os.path.join(save_dir, image_url.split('/')[-1])
        
        with open(file_name, 'wb') as f:
            for chunk in response.iter_content(chunk_size=8192):
                f.write(chunk)
                
        print(f'Image saved to {file_name}')
        
    except Exception as e:
        print(f'Error downloading image: {e}')

# Example usage

### PHP Example
This example illustrates how to use cURL in PHP to fetch and save an image from the ShootProof HD Image Downloader.

php
<?php

function download_image($image_url, $save_dir = 'downloads') {
    if (!file_exists($save_dir)) {
        mkdir($save_dir, 0777, true);
    }
    
    $file_name = $save_dir . '/' . basename($image_url);
    $ch = curl_init($image_url);

    curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
    curl_setopt($ch, CURLOPT_BINARYTRANSFER, true);
    
    $image_data = curl_exec($ch);
    $curl_error = curl_error($ch);
    
    if ($curl_error) {
        echo "Error downloading image: $curl_error";
    } else {
        file_put_contents($file_name, $image_data);
        echo "Image saved to $file_name";
    }
    
    curl_close($ch);
}

// Example usage
download_image('https://hdstockimages.com/shootproof-hd-image-downloader/image.jpg');
?>

async function downloadImage(imageUrl, saveDir = './downloads') {
    const fs = require('fs');
    const path = require('path');
    
    if (!fs.existsSync(saveDir)) {
        fs.mkdirSync(saveDir, { recursive: true });
    }

    try {
        const response = await fetch(imageUrl);
        
        if (!response.ok) {
            throw new Error(`HTTP error! status: ${response.status}`);
        }
        
        const buffer = await response.buffer();
        const fileName = path.join(saveDir, path.basename(imageUrl));
        
        fs.writeFileSync(fileName, buffer);
        console.log(`Image saved to ${fileName}`);
        
    } catch (error) {
        console.error(`Error downloading image: ${error.message}`);
    }
}

// Example usage
downloadImage('https://hdstockimages.com/shootproof-hd-image-downloader/image.jpg');
